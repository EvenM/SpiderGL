 12 /**
 13  * Gets default value for varibles.
 14  *
 15  * This function is used to get default values for optional variables.
 16  * If arg is undefined or is {@link SpiderGL.Core.DEFAULT}, then defaultValue will be returned. Otherwise, arg will be returned.
 17  *
 18  * @returns {any} Returns arg if arg is not undefined and is not {@link SpiderGL.Core.DEFAULT}, otherwise returns defaultValue.
 19  *
 20  * @example
 21  * var DEFAULT_V = 1;
 22  *
 23  * var v = null;
 24  * v = SpiderGL.Utility.getDefaultValue(someVar, DEFAULT_V); // someVar is undefined, so v = DEFAULT_V
 25  *
 26  * var someVar = 2;
 27  * v = SpiderGL.Utility.getDefaultValue(someVar, DEFAULT_V); // someVar is defined, so v = someVar
 28  *
 29  * var someVar = SpiderGL.Core.DEFAULT;
 30  * v = SpiderGL.Utility.getDefaultValue(someVar, DEFAULT_V); // someVar is SpiderGL.Core.DEFAULT, so v = DEFAULT_V
 31  *
 32  * @example
 33  * var DEFAULT_Y = 1;
 34  * var DEFAULT_Z = 2;
 35  *
 36  * function setObject(obj, x, options) {
 37  *   options = options || { }; // create an empty object to avoid testing for null
 38  *   obj.x = x;
 39  *   obj.y = SpiderGL.Utility.getDefaultValue(options.y, DEFAULT_Y);
 40  *   obj.z = SpiderGL.Utility.getDefaultValue(options.z, DEFAULT_Z);
 41  * }
 42  *
 43  * var obj = {
 44  *   x : 0,
 45  *   y : 1,
 46  *   z : 2
 47  * };
 48  *
 49  * setObject(obj, 3);        // obj = { x:3, y:DEFAULT_Y, z:DEFAULT_Z }
 50  * setObject(obj, 4, {z:5}); // obj = { x:4, y:DEFAULT_Y, z:5         }
 51  */
 52 SpiderGL.Utility.getDefaultValue = function (arg, defaultValue) {
 53 	if ((arg === undefined) || (arg === SpiderGL.Core.DEFAULT)) { return defaultValue; }
 54 	return arg;
 55 }

 82 /**
 83  * Sets default values for the passed object.
 84  *
 85  * @param {object} defaultObj The object containing the default values.
 86  * @param {object} obj The object from which values are extracted.
 87  *
 88  * @returns {object} The modified obj. If obj is null, defaultObj will be returned.
 89  */
 90 SpiderGL.Utility.setDefaultValues = function (defaultObj, obj) {
 91 	if (!obj) return defaultObj;
 92 
 93 	var sDefault = SpiderGL.Core.DEFAULT;
 94 	for (var p in obj) {
 95 		if (obj[p] == sDefault) {
 96 			if (typeof defaultObj[p] != "undefined") {
 97 				obj[p] = defaultObj[p];
 98 			}
 99 		}
100 	}
101 	for (var p in defaultObj) {
102 		if (typeof obj[p] == "undefined") {
103 			obj[p] = defaultObj[p];
104 		}
105 	}
106 	return obj;
107 };

109 /**
110  * Converts the input arguments to a 4-component Float32Array.
111  * The input value is handled like WebGL handles constant vertex attributes,
112  * that is, if the input parameter is null, a number, or an array with less than four components,
113  * missing values are taken from the array [0, 0, 0, 1] at the respective position. 
114  *
115  * @param {null|undefined|number|array|Float32Array} x The input value.
116  *
117  * @returns {array} a 4-component array.
118  *
119  * @example
120  * x = SpiderGL.Utility.getAttrib4fv();                        // x = [0, 0, 0, 1]
121  * x = SpiderGL.Utility.getAttrib4fv(null);                    // x = [0, 0, 0, 1]
122  * x = SpiderGL.Utility.getAttrib4fv(undefined);               // x = [0, 0, 0, 1]
123  * x = SpiderGL.Utility.getAttrib4fv(0);                       // x = [0, 0, 0, 1]
124  * x = SpiderGL.Utility.getAttrib4fv(7);                       // x = [7, 0, 0, 1]
125  * x = SpiderGL.Utility.getAttrib4fv([]);                      // x = [0, 0, 0, 1]
126  * x = SpiderGL.Utility.getAttrib4fv([1]);                     // x = [1, 0, 0, 1]
127  * x = SpiderGL.Utility.getAttrib4fv([1, 2]);                  // x = [1, 2, 0, 1]
128  * x = SpiderGL.Utility.getAttrib4fv([1, 2, 3]);               // x = [1, 2, 3, 1]
129  * x = SpiderGL.Utility.getAttrib4fv([1, 2, 3, 4, 5, 6]);      // x = [1, 2, 3, 4]
130  * x = SpiderGL.Utility.getAttrib4fv(new Float32Array([0,9]);  // x = [0, 9, 0, 1]
131  */
132 SpiderGL.Utility.getAttrib4fv = function (x) {
133 	if (SpiderGL.Type.isNumber(x)) return [x, 0, 0, 1];
134 	if (!x) return [0, 0, 0, 1];
135 	return [
136 		(x[0] != undefined) ? x[0] : 0,
137 		(x[1] != undefined) ? x[1] : 0,
138 		(x[2] != undefined) ? x[2] : 0,
139 		(x[3] != undefined) ? x[3] : 1
140 	];
141 }

143 /**
144  * Gets the number of milliseconds elapsed since January 1, 1970 at 00:00.
145  * It is a utility function for (new Date()).getTime().
146  *
147  * @returns {number} The number of milliseconds elapsed since January 1, 1970 at 00:00.
148  */
149 SpiderGL.Utility.getTime = function () {
150 	return (new Date()).getTime();

153 SpiderGL.Utility.Timer = function () {
154 	this._tStart   = -1;
155 	this._tElapsed = 0;
156 }
157 
158 SpiderGL.Utility.Timer.prototype = {
159 	_accumElapsed : function () {
160 		this._tElapsed += this.now - this._tStart;
161 	},
162 
163 	get now() {
164 		return Date.now();
165 	},
166 
167 	start : function () {
168 		if (this.isStarted) return;
169 		if (this.isPaused)  return;
170 		this._tStart   = this.now;
171 		this._tElapsed = 0;
172 	},
173 
174 	restart : function () {
175 		var r = this.elapsed;
176 		this._tStart   = this.now;
177 		this._tElapsed = 0;
178 		return r;
179 	},
180 
181 	stop : function () {
182 		if (!this.isStarted) return;
183 		if (this.isPaused)   return;
184 		this._accumElapsed();
185 		this._tStart = -1;
186 	},
187 
188 	get isStarted() {
189 		return (this._tStart >= 0);
190 	},
191 
192 	pause : function () {
193 		if (!this.isStarted) return;
194 		if (this.isPaused)   return;
195 		this._accumElapsed();
196 		this._tStart = -2;
197 	},
198 
199 	resume : function () {
200 		if (!this.isStarted) return;
201 		if (!this.isPaused)  return;
202 		this._tStart = this.now;
203 	},
204 
205 	get isPaused() {
206 		return (this._tStart == -2);
207 	},
208 
209 	get elapsed() {
210 		return ((this.isStarted) ? (this._tElapsed + (this.now - this._tStart)) : (this._tElapsed))
211 	}
212 };
