 12 /**
 13  * Utility tag for any use.
 14  *
 15  * @type any
 16  */
 17 SpiderGL.TAG = 0;

 19 /**
 20  * Publishes SpiderGL modules content to a global object.
 21  *
 22  * The effect is to pollute the global object with the SpiderGL symbols. At the same time, it allows a faster and less verbose code.
 23  *
 24  * @param {object} [options] Symbols publishing options.
 25  * @param {object} [options.globalObject=SpiderGL.openNamespace.DEFAULT_GLOBAL_OBJECT] The global object to which inject SpiderGL symbols.
 26  * @param {string} [options.constantPrefix=SpiderGL.openNamespace.DEFAULT_CONSTANT_PREFIX] String prefix for constants.
 27  * @param {string} [options.functionPrefix=SpiderGL.openNamespace.DEFAULT_FUNCTION_PREFIX] String prefix for functions. The first letter of the function identifier will be made capital.
 28  * @param {string} [options.classPrefix=SpiderGL.openNamespace.DEFAULT_CLASS_PREFIX] String prefix for classes and sub-modules.
 29  */
 30 SpiderGL.openNamespace = function (options) {
 31 	options = SpiderGL.Utility.getDefaultObject({
 32 		globalObject   : SpiderGL.openNamespace.DEFAULT_GLOBAL_OBJECT,
 33 		constantPrefix : SpiderGL.openNamespace.DEFAULT_CONSTANT_PREFIX,
 34 		functionPrefix : SpiderGL.openNamespace.DEFAULT_FUNCTION_PREFIX,
 35 		classPrefix    : SpiderGL.openNamespace.DEFAULT_CLASS_PREFIX
 36 	}, options);
 37 
 38 	var constantRegExp = new RegExp("^(([_\$0-9A-Z])+)$");
 39 	function isConstant(name) {
 40 		return constantRegExp.test(name);
 41 	}
 42 
 43 	var namespaceOrClassRegExp = new RegExp("^([A-Z])");
 44 	function isNamespaceOrClass(name) {
 45 		return (namespaceOrClassRegExp.test(name) && !isConstant(name));
 46 	}
 47 
 48 	var functionRegExp = new RegExp("^(([a-z])+([_\$0-9A-Za-z])*)$");
 49 	function isFunction(name) {
 50 		return functionRegExp.test(name);
 51 	}
 52 
 53 	function initialCap(name) {
 54 		return (name.substr(0, 1).toUpperCase() + name.substr(1));
 55 	}
 56 
 57 	var classes    = { };
 58 	var functions  = { };
 59 	var constants  = { };
 60 
 61 	function collect(module) {
 62 		if (!module) return;
 63 		for (var x in module) {
 64 			if (x.substr(0, 1) == "_") continue;
 65 			var y = module[x];
 66 			if (isNamespaceOrClass(x)) {
 67 				//if (classes[x]) log("Duplicate: " + x);
 68 				classes[x] = y;
 69 			}
 70 			else if (isFunction(x)) {
 71 				//if (functions[x]) log("Duplicate: " + x);
 72 				functions[x] = y;
 73 			}
 74 			else if (isConstant(x)) {
 75 				//if (constants[x]) log("Duplicate: " + x);
 76 				constants[x] = y;
 77 			}
 78 			else {
 79 				//log("Unknown : " + x);
 80 			}
 81 		}
 82 	}
 83 
 84 	var modules = [
 85 		"Core",
 86 		"DOM",
 87 		"IO",
 88 		"Math",
 89 		"Mesh",
 90 		"Model",
 91 		"Semantic",
 92 		"Space",
 93 		"Type",
 94 		"UserInterface",
 95 		"Utility",
 96 		"Version",
 97 		"WebGL"
 98 	];
 99 
100 	for (var x in modules) {
101 		collect(SpiderGL[modules[x]]);
102 	}
103 
104 	for (var x in classes) {
105 		var name = options.classPrefix + initialCap(x);
106 		//log("CLASS    : " + name);
107 		options.globalObject[name] = classes[x];
108 	}
109 
110 	for (var x in functions) {
111 		var name = options.functionPrefix + initialCap(x);
112 		//log("FUNCTION : " + name);
113 		options.globalObject[name] = functions[x];
114 	}
115 
116 	for (var x in constants) {
117 		var name = options.constantPrefix + initialCap(x);
118 		//log("CONSTANT : " + name);
119 		options.globalObject[name] = constants[x];
120 	}
121 };
