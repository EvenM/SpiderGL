 33 /**
 34  * The SpiderGL.Version namespace.
 35  *
 36  * @namespace The SpiderGL.Version namespace.
 37  */
 38 SpiderGL.Version = { };
 39 
 40 /**
 41  * Major version number.
 42  *
 43  * @constant
 44  * @type number
 45  */
 46 SpiderGL.Version.VERSION_MAJOR = 0;
 47 
 48 /**
 49  * Minor version number.
 50  *
 51  * @constant
 52  * @type number
 53  */
 54 SpiderGL.Version.VERSION_MINOR = 2;
 55 
 56 /**
 57  * Revision version number.
 58  *
 59  * @constant
 60  * @type number
 61  */
 62 SpiderGL.Version.VERSION_REVISION = 0;
 63 
 64 /**
 65  * Version string.
 66  *
 67  * The version string is: "{@link SpiderGL.Version.VERSION_MAJOR}.{@link SpiderGL.Version.VERSION_MINOR}.{@link SpiderGL.Version.VERSION_REVISION}"
 68  *
 69  * @constant
 70  * @type string
 71  */
 72 SpiderGL.Version.VERSION_STRING = SpiderGL.Version.VERSION_MAJOR + "." + SpiderGL.Version.VERSION_MINOR + "." + SpiderGL.Version.VERSION_REVISION;
