 82 /**
 83  * Generates a unique id.
 84  *
 85  * @returns {number} A unique id.
 86  */
 87 SpiderGL.Core.generateUID = function () {
 88 	SpiderGL.Core.generateUID._lastUID++;
 89 	return SpiderGL.Core.generateUID._lastUID;
 90 }
 91 
 92 SpiderGL.Core.generateUID._lastUID = 0;
 93 
 94 /**
 95  * Creates a SpiderGL.Core.ObjectBase
 96  *
 97  * SpiderGL.Core.ObjectBase is the base class for all SpiderGL classes.
 98  *
 99  * @class The SpiderGL.Core.ObjectBase is the base class for all SpiderGL classes.
100  */
101 SpiderGL.Core.ObjectBase = function () {
102 	this._uid = SpiderGL.Core.generateUID();
103 };
