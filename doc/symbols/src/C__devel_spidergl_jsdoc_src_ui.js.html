 12 /**
 13  * Creates a SpiderGL.UserInterface.CanvasHandler.
 14  *
 15  * SpiderGL.UserInterface.CanvasHandler is an event handler used to implement an easy controller for WebGL canvas.
 16  * It should not be directly used. Use {@link SpiderGL.UserInterface.handleCanvas} or {@link SpiderGL.UserInterface.handleCanvasOnLoad} to install it to a listener object.
 17  *
 18  * @class The SpiderGL.UserInterface.CanvasHandler is an event handler used to implement an easy controller for WebGL canvas. It should not be directly used.
 19  *
 20  * @augments SpiderGL.Core.ObjectBase
 21  *
 22  * @param {WebGLRenderingcontext} gl The WebGLRenderingcontext to pass to the handler.
 23  * @param {object} handler The event handler to which events will be dispatched.
 24  * @param {object} [options] Optional parameters
 25  * @param {bool} [options.standardGLUnpack=SpiderGL.UserInterface.CanvasHandler.DEFAULT_STANDARD_GL_UNPACK] If true, sets the default OpenGL unpack behaviour.
 26  * @param {number} [options.animateRate=SpiderGL.UserInterface.CanvasHandler.DEFAULT_ANIMATE_RATE] Additional options.
 27  *
 28  * @see SpiderGL.UserInterface.handleCanvas
 29  * @see SpiderGL.UserInterface.handleCanvasOnLoad
 30  */

 76 	/** @private */
 77 	var handleMessage = function (evt) {
 78 		if (evt.source != window) return;
 79 		if (evt.data == SpiderGL.UserInterface.CanvasHandler._FAST_ANIMATE_MESSAGE_NAME) {
 80 			evt.stopPropagation();
 81 			that._onAnimate();
 82 		}
 83 		else if (evt.data == SpiderGL.UserInterface.CanvasHandler._FAST_DRAW_MESSAGE_NAME) {
 84 			evt.stopPropagation();
 85 			that._onDraw();
 86 		}
 87 	};
 88 	window.addEventListener("message", handleMessage, true);
 89  
 90 	canvas.contentEditable = true;
 91 	canvas.tabIndex = 0;
 92 
 93 	canvas.addEventListener("unload",          function (e) { that._onTerminate       (e); }, false);
 94 	canvas.addEventListener("keydown",         function (e) { that._onKeyDown         (e); }, false);
 95 	canvas.addEventListener("keyup",           function (e) { that._onKeyUp           (e); }, false);
 96 	canvas.addEventListener("keypress",        function (e) { that._onKeyPress        (e); }, false);
 97 	canvas.addEventListener("mousedown",       function (e) { that._onMouseButtonDown (e); }, false);
 98 	canvas.addEventListener("mouseup",         function (e) { that._onMouseButtonUp   (e); }, false);
 99 	canvas.addEventListener("mousemove",       function (e) { that._onMouseMove       (e); }, false);
100 	canvas.addEventListener("mouseout",        function (e) { that._onMouseOut        (e); }, false);
101 	canvas.addEventListener("click",           function (e) { that._onClick           (e); }, false);
102 	canvas.addEventListener("dblclick",        function (e) { that._onDoubleClick     (e); }, false);
103 	canvas.addEventListener("resize",          function (e) { that._onResize          (e); }, false);
104 	canvas.addEventListener("DOMMouseScroll",  function (e) { that._onMouseWheel      (e); }, false);
105 	canvas.addEventListener("mousewheel",      function (e) { that._onMouseWheel      (e); }, false);
106 	canvas.addEventListener("blur",            function (e) { that._onBlur            (e); }, false);
107 
108 	window.addEventListener("mouseup",         function (e) { that._onWindowMouseButtonUp (e); }, false);
109 	window.addEventListener("mousemove",       function (e) { that._onWindowMouseMove     (e); }, false);
110 
111 	canvas.addEventListener("touchstart",      SpiderGL.UserInterface.CanvasHandler._touchHandler, true);
112 	canvas.addEventListener("touchend",        SpiderGL.UserInterface.CanvasHandler._touchHandler, true);
113 	canvas.addEventListener("touchmove",       SpiderGL.UserInterface.CanvasHandler._touchHandler, true);
114 	canvas.addEventListener("touchcancel",     SpiderGL.UserInterface.CanvasHandler._touchHandler, true);
115 
116 	var standardGLUnpack = SpiderGL.Utility.getDefaultValue(options.standardGLUnpack, SpiderGL.UserInterface.CanvasHandler.DEFAULT_STANDARD_GL_UNPACK);
117 	if (standardGLUnpack) {
118 		SpiderGL.WebGL.Context.setStandardGLUnpack(gl);
119 	}
120 
121 	this.animateRate = SpiderGL.Utility.getDefaultValue(options.animateRate, SpiderGL.UserInterface.CanvasHandler.DEFAULT_ANIMATE_RATE);
122 }

163 SpiderGL.UserInterface.CanvasHandler._touchHandler = function (event) {
164 	var touches = event.changedTouches,
165 
166 	first = touches[0],
167 	type = "";
168 
169 	switch(event.type)
170 	{
171 		case "touchstart": type = "mousedown"; break;
172 		case "touchmove":  type = "mousemove"; break;        
173 		case "touchend":   type = "mouseup";   break;
174 		default: return;
175 	}
176 
177 	//initMouseEvent(type, canBubble, cancelable, view, clickCount, 
178 	//           screenX, screenY, clientX, clientY, ctrlKey, 
179 	//           altKey, shiftKey, metaKey, button, relatedTarget);
180 
181 	var simulatedEvent = document.createEvent("MouseEvent");
182 	simulatedEvent.initMouseEvent(
183 		type, true, true, window, 1, 
184 		first.screenX, first.screenY, 
185 		first.clientX, first.clientY, false, 
186 		false, false, false, 0/*left*/, null
187 	);
188 
189 	first.target.dispatchEvent(simulatedEvent);
190 	event.preventDefault();
191 };

220 	_getMouseClientPos : function(e) {
221 		var r = this._canvas.getBoundingClientRect();
222 		var w = this._canvas.width;
223 		var h = this._canvas.height;
224 		var x = e.clientX - r.left;
225 		var y = h - 1 - (e.clientY - r.top);
226 		var outside = ((x < 0) || (x >= w) || (y < 0) || (y >= h));
227 		return [x, y, outside];
228 	},

313 	_onMouseButtonDown : function (e) {
314 		this._canvas.focus();
315 		var xy  = this._getMouseClientPos(e);
316 		var btn = e.button;
317 		this._mouseButtonsDown[btn] = true;
318 		this._dragStartPos[btn] = [xy[0], xy[1]];

322 	},

324 	_onMouseButtonUp : function (e) {
325 		var xy = this._getMouseClientPos(e);
326 		var btn = e.button;
327 		this._mouseButtonsDown[btn] = false;
328 		this._dispatch("onMouseButtonUp", btn, xy[0], xy[1], e);
329 
330 		if (this._dragging[btn]) {
331 			this._dragging[btn] = false;
332 			var sPos = this._dragStartPos[btn];
333 			var ePos = [xy[0], xy[1]];
334 			this._dragEndPos[btn] = ePos;
335 			this._dragDeltaPos[btn] = [ePos[0] - sPos[0], ePos[1] - sPos[1]];
336 			this._dispatch("onDragEnd", btn, ePos[0], ePos[1]);
337 		}

372 			var sPos = this._dragStartPos[i];
373 			var ePos = [xy[0], xy[1]];

375 			this._dragDeltaPos[i] = [ePos[0] - sPos[0], ePos[1] - sPos[1]];

377 				this._dragging[i] = true;
378 				this._dispatch("onDragStart", i, sPos[0], sPos[1]);
379 			}
380 			this._dispatch("onDrag", i, ePos[0], ePos[1]);
381 		}
382

448 		this._animatePrevTime  = this._animateTime;
449 		this._animateTime      = Date.now();
450 		this._animateDeltaTime = this._animateTime - this._animatePrevTime;
451 		this._dispatch("onAnimate", this._animateDeltaTime / 1000);
452 		if (this._animateMS >= 0) {
453 			if (this._animateWithTimeout) {
454 				setTimeout(this._animateEventHandler, this._animateMS);
455 			}
456 		}
457 		else if (this._fastAnimate) {
458 			window.postMessage(SpiderGL.UserInterface.CanvasHandler._FAST_ANIMATE_MESSAGE_NAME, "*");
459 		}

462 	_onDraw : function () {
463 		this._drawEventPending = false;
464 
465 		this._fpsCount++;
466 
467 		var now   = Date.now();
468 		var fpsDT = now - this._fpsTime;
469 		if (fpsDT >= this._fpsUpdateMS) {
470 			this._fps      = SpiderGL.Math.floor(this._fpsCount * 1000 / fpsDT);
471 			this._fpsTime  = now;
472 			this._fpsCount = 0;
473 		}
474 
475 		this._dispatch("onDraw");
476 	},

579 	set animateRate(r) {
580 		r = SpiderGL.Utility.getDefaultValue(r, SpiderGL.UserInterface.CanvasHandler.DEFAULT_ANIMATE_RATE);
581 		if (this._animateRate === r) return;
582 
583 		this._fastAnimate = false;
584 		this._animateMS   = -1;
585 
586 		this._animateTime      = Date.now();
587 		this._animatePrevTime  = this._animateTime;
588 		this._animateDeltaTime = 0;
589 
590 		if (this._animateID) {
591 			clearInterval(this._animateID);
592 			this._animateID = null;
593 		}
594 
595 		if (r > 0) {
596 			this._animateMS = SpiderGL.Math.floor(1000 / r);
597 			if (this._animateWithTimeout) {
598 				setTimeout(this._animateEventHandler, this._animateMS);
599 			}
600 			else {
601 				this._animateID = setInterval(this._animateEventHandler, this._animateMS);
602 			}
603 		}
604 		else if (r < 0) {
605 			this._fastAnimate = true;
606 			window.postMessage(SpiderGL.UserInterface.CanvasHandler._FAST_ANIMATE_MESSAGE_NAME, "*");
607 		}
608 	},

802 /**
803  * Sets up a WebGL context and canvas event handling.
804  * The WebGLRenderingContext is created and hijacked using {@link SpiderGL.WebGL.Context.getHijacked}.
805  * A {@link SpiderGL.UserInterface.CanvasHandler} is created to handle the canvas events and dispatch them to the provided handler.
806  * The canvas handler will be installed in the handler object as a named property.
807  *
808  * @param {HTMLCanvasElement|string} canvas The canvas used for rendering and from which event will be received.
809  * @param {object} handler The event handler.
810  * @param {function()} [handler.onInitialize] onInitialize event handler.
811  * @param {function()} [handler.onTerminate] onTerminate event handler.
812  * @param {function(keyCode, event)} [handler.onKeyUp] onKeyUp event handler.
813  * @param {function(keyCode, event)} [handler.onKeyDown] onKeyDown event handler.
814  * @param {function(keyCode, event)} [handler.onKeyPress] onKeyPress event handler.
815  * @param {function(button, cursorX, cursorY, event)} [handler.onMouseButtonDown] onMouseButtonDown event handler.
816  * @param {function(button, cursorX, cursorY, event)} [handler.onMouseButtonUp] onMouseButtonUp event handler.
817  * @param {function(cursorX, cursorY, event)} [handler.onMouseMove] onMouseMove event handler.
818  * @param {function(wheelDelta, cursorX, cursorY, event)} [handler.onMouseWheel] onMouseWheel event handler.
819  * @param {function(button, cursorX, cursorY, event)} [handler.onClick] onClick event handler.
820  * @param {function(button, cursorX, cursorY, event)} [handler.onDoubleClick] onDoubleClick event handler.
821  * @param {function(button, cursorX, cursorY)} [handler.onDragStart] onDragStart event handler.
822  * @param {function(button, cursorX, cursorY)} [handler.onDragEnd] onDragEnd event handler.
823  * @param {function(button, cursorX, cursorY)} [handler.onDrag] onDrag event handler.
824  * @param {function(canvasWidth, canvasHeight)} [handler.onResize] onResize event handler.
825  * @param {function(deltaTimeSecs)} [handler.onAnimate] onAnimate event handler.
826  * @param {function()} [handler.onDraw] onDraw event handler.
827  * @param {object} options Optional parameters (see {@link SpiderGL.UserInterface.CanvasHandler}).
828  * @param {string} [options.uiName=SpiderGL.UserInterface.CanvasHandler.DEFAULT_PROPERTY_NAME] The name of the property to install in the handler object for accessing the canvas handler.
829  *
830  * @see SpiderGL.UserInterface.handleCanvasOnLoad
831  * @see SpiderGL.UserInterface.CanvasHandler
832  */
833 SpiderGL.UserInterface.handleCanvas = function (canvas, handler, options) {
834 	if (!canvas || !handler) return false;
835 
836 	options = options || { };
837 
838 	var gl = SpiderGL.WebGL.Context.getHijacked(canvas, options);
839 	if (!gl) return false;
840 
841 	var ui = new SpiderGL.UserInterface.CanvasHandler(gl, handler, options);
842 	if (!ui) return false;
843 
844 	var uiName = SpiderGL.Utility.getDefaultValue(options.uiName, SpiderGL.UserInterface.CanvasHandler.DEFAULT_PROPERTY_NAME);
845 	handler[uiName] = ui;
846 	ui._firstNotify();
847 
848 	return true;
849 }

851 /**
852  * Sets up a WebGL context and canvas event handling after window loading.
853  * When the window fires the onload event, {@link SpiderGL.UserInterface.handleCanvas} is called.
854  *
855  * @param {HTMLCanvasElement|string} canvas The canvas used for rendering and from which event will be received.
856  * @param {object} handler The event handler (see {@link SpiderGL.UserInterface.handleCanvas}).
857  * @param {object} options Optional parameters (see {@link SpiderGL.UserInterface.handleCanvas}).
858  *
859  * @see SpiderGL.UserInterface.handleCanvas
860  */
861 SpiderGL.UserInterface.handleCanvasOnLoad = function (canvas, handler, options) {
862 	if (!canvas || !handler) return false;
863 
864 	options = options || { };
865 	var onLoad = SpiderGL.Utility.getDefaultValue(options.onLoad, null);
866 
867 	function doHandle() {
868 		SpiderGL.UserInterface.handleCanvas(canvas, handler, options);
869 		if (onLoad) { onLoad(); }
870 	}
871 
872 	window.addEventListener("load", doHandle, false);
873 	return true;
874 }
