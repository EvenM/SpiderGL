 12 /**
 13  * Creates a SpiderGL.Model.Model.
 14  *
 15  * @class The SpiderGL.Model.Model represents a complex geometric model.
 16  *
 17  * @augments SpiderGL.Core.ObjectBase
 18  */
 19 SpiderGL.Model.Model = function (gl, descriptor, options) {
 20 	SpiderGL.Core.ObjectBase.call(this);
 21 
 22 	options = SpiderGL.Utility.getDefaultObject({
 23 	}, options);
 24 
 25 	if (descriptor && ("vertices" in descriptor)) {
 26 		descriptor = SpiderGL.Model.Model._createSimpleDescriptor(descriptor);
 27 	}
 28 
 29 	this._descriptor = SpiderGL.Model.Model._fixDescriptor(descriptor);
 30 	this._gl = null;
 31 	this._renderData = { };
 32 
 33 	if (gl) {
 34 		this.updateGL(gl, options);
 35 		this.updateRenderData();
 36 	}
 37 };

400 SpiderGL.Model.Model._createSimpleDescriptor = function (options) {
401 	options = SpiderGL.Utility.getDefaultObject({
402 		vertices   : null,
403 		primitives : null,
404 		options    : null
405 	}, options);
406 
407 	var bindingName        = "mainBinding";
408 	var chunkName          = "mainChunk";
409 	var partName           = "mainPart";
410 	var vertexBufferSuffix = "VertexBuffer";
411 	var indexBufferSuffix  = "IndexBuffer";
412 
413 	var d = {
414 		data : {
415 			vertexBuffers : {
416 			},
417 			indexBuffers : {
418 			},
419 		},
420 		access : {
421 			vertexStreams : {
422 			},
423 			primitiveStreams : {
424 			}
425 		},
426 		semantic : {
427 			bindings : {
428 			},
429 			chunks : {
430 			}
431 		},
432 		logic : {
433 			parts : {
434 			}
435 		}
436 	};
437 
438 	var binding = {
439 		vertexStreams : {
440 		},
441 		primitiveStreams : {
442 		}
443 	};
444 	d.semantic.bindings[bindingName] = binding;
445 
446 	var chunk = {
447 		techniques : {
448 			"common" : {
449 				binding : bindingName
450 			}
451 		}
452 	};
453 	d.semantic.chunks[chunkName] = chunk;
454 
455 	var part = {
456 		chunks : [ chunkName ]
457 	};
458 	d.logic.parts[partName] = part;
459 
460 	var minBufferedCount = -1;
461 	var hasBuffered = false;
462 	var hasConstant = false;
463 
464 	for (var x in options.vertices) {
465 		var src = options.vertices[x];
466 		if (!src) continue;
467 
468 		if (SpiderGL.Type.isArray(src) || SpiderGL.Type.isTypedArray(src) || SpiderGL.Type.instanceOf(src, ArrayBuffer)) {
469 			src = { data : src };
470 		}
471 
472 		var map = SpiderGL.Model.Model.DEFAULT_SIMPLE_MODEL_VERTEX_MAP[x];
473 		var mapSemantic = null;
474 		if (!map) {
475 			map = SpiderGL.Model.Model.DEFAULT_SIMPLE_MODEL_VERTEX_MAP["user"];
476 			mapSemantic = x.toUpperCase();
477 		}
478 		else {
479 			mapSemantic = map.semantic;
480 		}
481 
482 		var info = SpiderGL.Utility.getDefaultObject({
483 			size       : map.size,
484 			type       : map.type,
485 			normalized : map.normalized,
486 			semantic   : mapSemantic,
487 			index      : map.index,
488 			data       : null,
489 			value      : map.value.slice()
490 		}, src);
491 
492 		var accessor = {
493 			buffer     : null,
494 			size       : info.size,
495 			type       : info.type,
496 			normalized : info.normalized,
497 			stride     : 0,
498 			offset     : 0,
499 			value      : info.value.slice(),
500 		};
501 
502 		if (info.data) {
503 			var buffer = {
504 				type : info.type
505 			};
506 			var count = 0;
507 			if (SpiderGL.Type.isArray(info.data)) {
508 				buffer.untypedArray = info.data;
509 				count = buffer.untypedArray.length / accessor.size;
510 			}
511 			else if (SpiderGL.Type.isTypedArray(src) || SpiderGL.Type.instanceOf(src, ArrayBuffer)) {
512 				buffer.typedArray = info.data;
513 				count = (buffer.typedArray.byteLength - accessor.offset) / (accessor.size * SpiderGL.Type.typeSize(accessor.type));
514 			}
515 			else {
516 				continue;
517 			}
518 			hasBuffered = true;
519 			minBufferedCount = (minBufferedCount >= 0) ? (SpiderGL.Math.min(minBufferedCount, count)) : (count);
520 			var bufferName = x + vertexBufferSuffix;
521 			d.data.vertexBuffers[bufferName] = buffer;
522 			accessor.buffer = bufferName;
523 		}
524 		else {
525 			hasConstant = true;
526 		}
527 
528 		var streamName = x;
529 		d.access.vertexStreams[streamName] = accessor;
530 
531 		var streams = new Array(info.index + 1);
532 		streams[info.index] = streamName;
533 		binding.vertexStreams[info.semantic] = streams;
534 	}
535 
536 	var minCount = 0;
537 	if (hasBuffered) {
538 		minCount = minBufferedCount;
539 	}
540 	else if (hasConstant) {
541 		minCount = 1;
542 	}
543 
544 	var optionsPrimitives = options.primitives;
545 	if (SpiderGL.Type.isString(optionsPrimitives)) {
546 		optionsPrimitives = [ optionsPrimitives ];
547 	}
548 	if (SpiderGL.Type.isArray(optionsPrimitives)) {
549 		var op = optionsPrimitives;
550 		optionsPrimitives = { };
551 		for (var i=0, n=op.length; i<n; ++i) {
552 			var pn = op[i];
553 			if (!SpiderGL.Model.Model.DEFAULT_SIMPLE_MODEL_PRIMITIVE_MAP[pn]) continue;
554 			optionsPrimitives[pn] = { };
555 		}
556 	}
557 
558 	for (var x in optionsPrimitives) {
559 		var src = optionsPrimitives[x];
560 		if (!src) continue;
561 
562 		if (SpiderGL.Type.isArray(src) || SpiderGL.Type.isTypedArray(src) || SpiderGL.Type.instanceOf(src, ArrayBuffer)) {
563 			src = { data : src };
564 		}
565 
566 		var map = SpiderGL.Model.Model.DEFAULT_SIMPLE_MODEL_PRIMITIVE_MAP[x];
567 		if (!map) map = SpiderGL.Model.Model.DEFAULT_SIMPLE_MODEL_PRIMITIVE_MAP["user"];
568 
569 		var info = SpiderGL.Utility.getDefaultObject({
570 			mode     : map.mode,
571 			type     : map.type,
572 			count    : ((map.count >= 0) ? (map.count) : (minCount)),
573 			semantic : map.semantic
574 		}, src);
575 
576 		var accessor = {
577 			buffer     : null,
578 			mode       : info.mode,
579 			first      : 0,
580 			count      : info.count,
581 			type       : info.type,
582 			offset     : 0
583 		};
584 
585 		if (info.data) {
586 			var buffer = { };
587 			if (SpiderGL.Type.isArray(info.data)) {
588 				buffer.untypedArray = info.data;
589 			}
590 			else if (SpiderGL.Type.isTypedArray(src) || SpiderGL.Type.instanceOf(src, ArrayBuffer)) {
591 				buffer.typedArray = info.data;
592 			}
593 			else {
594 				continue;
595 			}
596 			var bufferName = x + indexBufferSuffix;
597 			d.data.indexBuffers[bufferName] = buffer;
598 			accessor.buffer = bufferName;
599 		}
600 
601 		var streamName = x;
602 		d.access.primitiveStreams[streamName] = accessor;
603 
604 		var streams = new Array(1);
605 		streams[0] = streamName;
606 		binding.primitiveStreams[info.semantic] = streams;
607 	}
608 
609 	return d;
610 };

629 	updateTypedArrays : function () {
630 		var d = this._descriptor;
631 		if (!d) return false;
632 
633 		var buffer = null;
634 		var ctor   = null;
635 
636 		var vertexBuffers = d.data.vertexBuffers;
637 		for (var x in vertexBuffers) {
638 			buffer = vertexBuffers[x];
639 			if (!buffer.untypedArray) continue;
640 			ctor = SpiderGL.Type.typeToTypedArrayConstructor(buffer.type);
641 			buffer.typedArray = new ctor(buffer.untypedArray);
642 		}
643 
644 		var indexBuffers = d.data.indexBuffers;
645 		for (var x in indexBuffers) {
646 			buffer = indexBuffers[x];
647 			if (!buffer.untypedArray) continue;

881 SpiderGL.Model.Technique._fixVertexStreams = function (p, d) {
